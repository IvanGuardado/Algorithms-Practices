\begin{algorithm}
\begin{algorithmic}
\FOR{$i \gets 1$ to $length[A]-1$}
\STATE {$min \gets i$}
\FOR{$j \gets i+1$ to $length[A]$}
\IF{$A[j]<A[min]$}
\STATE {$min \gets j$}
\ENDIF
\ENDFOR
\STATE $aux \gets A[i]$
\STATE $A[i] \gets A[min]$
\STATE $A[min] \gets aux$
\ENDFOR
\end{algorithmic}
\caption{Selection Sort(A)}
\end{algorithm}
